def read_sentence(sentence, stopwords, corpus, sentences):
    try:
        segment = jieba.lcut(sentence.strip())
        segment = [word for word in segment if (word not in stopwords.values and word != ' ' and word != '\n')] # 
        corpus.extend(segment)
        sentences.append(segment)
    except Exception as e:
        print(e, '\n', sentence)
    finally:
        return(segment)

def get_wordcloud(corpus):
    '''
    type of corpus is list
    '''
    corpus_df = pd.DataFrame({'word': corpus})
    corpus_df = corpus_df.groupby('word')['word'].agg({'size': np.size})
    corpus_df = corpus_df.reset_index().sort_values(by=['size'],ascending=False)
    wordcloud = WordCloud(font_path="simhei.ttf",background_color="white",max_font_size=80)
    word_frequence = {x[0]:x[1] for x in corpus_df.head(corpus_df.shape[0]//5).values}
    wordcloud = wordcloud.fit_words(word_frequence)
    matplotlib.rcParams['figure.figsize'] = (14.0, 5.0)
    plt.imshow(wordcloud)
